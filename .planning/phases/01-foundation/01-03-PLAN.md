```python
from pathlib import Path
from langchain_core.documents import Document
from langchain_text_splitters import RecursiveCharacterTextSplitter
import config

def load_kb_articles() -> list[Document]:
    kb_dir = config.BASE_DIR / "docs" / "kb_snapshots"
    if not kb_dir.exists():
        import logging
        logging.getLogger(__name__).warning("KB snapshots dir not found: %s", kb_dir)
        return []
    splitter = RecursiveCharacterTextSplitter(
        chunk_size=config.CHUNK_SIZE,
        chunk_overlap=config.CHUNK_OVERLAP,
    )
    documents = []
    for md_file in sorted(kb_dir.glob("*.md")):
        text = md_file.read_text(encoding="utf-8", errors="replace").strip()
        if len(text) < 50:
            continue
        for i, chunk in enumerate(splitter.split_text(text)):
            documents.append(Document(
                page_content=chunk,
                metadata={
                    "source": f"kb_articles:{md_file.name}",
                    "source_url": f"kb_articles:{md_file.name}",
                    "source_type": "kb_articles",
                    "file_name": md_file.name,
                    "chunk_index": i,
                },
            ))
    return documents
```

The FedEx version is a generic file walker — adapt it as a standalone loader that returns list[Document] (not calling index_codebase directly). run_ingest.py will call index_codebase() for code sources (storepepsaas, automation) and call add_documents() for doc sources.

Actually the cleaner pattern is: codebase_loader.py provides load_codebase(path, source_type, extensions, exclude_dirs) → list[Document] (simple walker + chunker), and run_ingest.py calls add_documents(load_codebase(...)) for doc-like sources OR calls rag.code_indexer.index_codebase() for code sources.

For MCSL, route storepepsaas and automation through rag.code_indexer.index_codebase() (writes to mcsl_code_knowledge). codebase_loader.py is available as a fallback loader if needed.

Implement codebase_loader.py with:
```python
from pathlib import Path
from langchain_core.documents import Document
from langchain_text_splitters import RecursiveCharacterTextSplitter
import logging, os
import config

logger = logging.getLogger(__name__)

_DEFAULT_EXTENSIONS = {".ts", ".js", ".jsx", ".tsx", ".md", ".py"}
_DEFAULT_SKIP_DIRS = {"node_modules", ".git", "dist", "build", "__pycache__", ".venv", "venv", "coverage", "carrier-envs"}
_MAX_FILE_BYTES = 100_000

def load_codebase(
    path: str,
    source_type: str = "codebase",
    extensions: set | list | None = None,
    exclude_dirs: set | list | None = None,
) -> list[Document]:
    root = Path(path)
    if not root.exists():
        logger.warning("Codebase path does not exist: %s — skipping", root)
        return []
    exts = set(extensions) if extensions else _DEFAULT_EXTENSIONS
    skip = set(exclude_dirs) if exclude_dirs else _DEFAULT_SKIP_DIRS
    splitter = RecursiveCharacterTextSplitter(chunk_size=800, chunk_overlap=100)
    documents = []
    for dirpath, dirnames, filenames in os.walk(root):
        dirnames[:] = [d for d in dirnames if d not in skip]
        for fname in filenames:
            fpath = Path(dirpath) / fname
            if not any(fname.endswith(ext) for ext in exts):
                continue
            if fpath.stat().st_size > _MAX_FILE_BYTES or fpath.stat().st_size == 0:
                continue
            try:
                text = fpath.read_text(encoding="utf-8", errors="ignore")
                rel = str(fpath.relative_to(root))
                for i, chunk in enumerate(splitter.split_text(text)):
                    documents.append(Document(
                        page_content=f"// File: {rel}\n\n{chunk}",
                        metadata={"source_type": source_type, "file_path": rel, "source": "codebase", "chunk_index": i},
                    ))
            except Exception as e:
                logger.warning("Failed to read %s: %s", fpath, e)
    logger.info("codebase_loader: %d chunks from %s (%s)", len(documents), path, source_type)
    return documents
```

```python
#!/usr/bin/env python3
"""
MCSL Domain Expert — Master Ingestion Pipeline

Usage:
    python ingest/run_ingest.py                        # Ingest all 5 sources
    python ingest/run_ingest.py --sources kb_articles  # Only KB articles
    python ingest/run_ingest.py --sources wiki automation
"""
from __future__ import annotations
import argparse
import logging
import sys
import time
import config

logging.basicConfig(level=logging.INFO, format="%(levelname)s %(name)s: %(message)s")
logger = logging.getLogger(__name__)

_ALL_SOURCES = ["kb_articles", "sheets", "wiki", "storepepsaas", "automation"]

def run_ingest(sources: list[str] | None = None) -> None:
    from rag.vectorstore import delete_by_source_type, add_documents
    from rag.code_indexer import index_codebase
    from ingest.kb_loader import load_kb_articles
    from ingest.sheets_loader import load_test_cases
    from ingest.wiki_loader import load_wiki_docs

    active = sources if sources is not None else _ALL_SOURCES
    start = time.time()

    print("=" * 60)
    print("MCSL Domain Expert — Knowledge Base Ingestion")
    print(f"Sources: {', '.join(active)}")
    print("=" * 60)

    if "kb_articles" in active:
        logger.info("Ingesting KB articles...")
        docs = load_kb_articles()
        logger.info("KB articles: %d chunks loaded", len(docs))
        delete_by_source_type("kb_articles")
        add_documents(docs)

    if "sheets" in active:
        logger.info("Ingesting TC sheet (Google Sheets)...")
        docs = load_test_cases()
        logger.info("TC sheet: %d chunks loaded", len(docs))
        delete_by_source_type("sheets")
        add_documents(docs)

    if "wiki" in active:
        logger.info("Ingesting MCSL wiki...")
        docs = load_wiki_docs()
        logger.info("Wiki: %d chunks loaded", len(docs))
        delete_by_source_type("wiki")
        add_documents(docs)

    if "storepepsaas" in active:
        logger.info("Indexing storepepSAAS codebase (server/src/shared/)...")
        result = index_codebase(
            code_path=config.STOREPEPSAAS_SHARED_PATH,
            source_type="storepepsaas",
            extensions=[".js"],
            clear_existing=True,
        )
        logger.info("storepepSAAS: %s", result)

    if "automation" in active:
        logger.info("Indexing mcsl-test-automation repo...")
        result = index_codebase(
            code_path=config.MCSL_AUTOMATION_REPO_PATH,
            source_type="automation",
            extensions=[".ts", ".js", ".md"],
            clear_existing=True,
        )
        logger.info("automation: %s", result)

    elapsed = time.time() - start
    print(f"\nDone: all sources indexed in {elapsed:.1f}s")
    print("=" * 60)

if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="Rebuild MCSL Domain Expert knowledge base")
    parser.add_argument(
        "--sources",
        nargs="*",
        choices=_ALL_SOURCES,
        help="Which sources to ingest (default: all 5)",
    )
    args = parser.parse_args()
    run_ingest(args.sources)
```
